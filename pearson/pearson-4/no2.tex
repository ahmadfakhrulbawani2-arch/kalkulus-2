\section*{Problem Statement}
Estimate the area under the graph of the function $f(x) = 8 - x^2 + 2x$ between $x = -2$ and $x = 4$ using finite approximations for the following cases:
\begin{enumerate}
    \item Lower sum with two rectangles of equal width.
    \item Lower sum with four rectangles of equal width.
    \item Upper sum with two rectangles of equal width.
    \item Upper sum with four rectangles of equal width.
\end{enumerate}

\section*{Function Analysis}
The function is a downward-opening parabola:
\[ f(x) = -x^2 + 2x + 8 \]
The $x$-coordinate of the vertex is:
\[ x = \frac{-b}{2a} = \frac{-2}{2(-1)} = 1 \]
The $y$-coordinate (maximum value) is:
\[ f(1) = 8 - (1)^2 + 2(1) = 9 \]
Endpoints of the interval $[-2, 4]$:
\[ f(-2) = 8 - (-2)^2 + 2(-2) = 8 - 4 - 4 = 0 \]
\[ f(4) = 8 - (4)^2 + 2(4) = 8 - 16 + 8 = 0 \]

\section*{Solutions}

\subsection*{a. Lower sum with $n=2$ rectangles}
Interval width: $\Delta x = \frac{4 - (-2)}{2} = 3$. \\
Sub-intervals: $[-2, 1]$ and $[1, 4]$.
\begin{itemize}
    \item On $[-2, 1]$, the minimum value is $f(-2) = 0$.
    \item On $[1, 4]$, the minimum value is $f(4) = 0$.
\end{itemize}
\[ L_2 = \sum_{i=1}^{2} f_{min, i} \cdot \Delta x = (0 \cdot 3) + (0 \cdot 3) = 0 \]

\subsection*{b. Lower sum with $n=4$ rectangles}
Interval width: $\Delta x = \frac{4 - (-2)}{4} = 1.5$. \\
Sub-intervals: $[-2, -0.5], [-0.5, 1], [1, 2.5], [2.5, 4]$.
\begin{itemize}
    \item $[-2, -0.5]$: Min is $f(-2) = 0$.
    \item $[-0.5, 1]$: Min is $f(-0.5) = 8 - (-0.5)^2 + 2(-0.5) = 8 - 0.25 - 1 = 6.75$.
    \item $[1, 2.5]$: Min is $f(2.5) = 8 - (2.5)^2 + 2(2.5) = 8 - 6.25 + 5 = 6.75$.
    \item $[2.5, 4]$: Min is $f(4) = 0$.
\end{itemize}
\[ L_4 = 1.5 \times (0 + 6.75 + 6.75 + 0) = 1.5 \times 13.5 = 20.25 \]

\subsection*{c. Upper sum with $n=2$ rectangles}
Interval width: $\Delta x = 3$. \\
Sub-intervals: $[-2, 1]$ and $[1, 4]$.
\begin{itemize}
    \item On $[-2, 1]$, the maximum value is at the vertex $x=1$: $f(1) = 9$.
    \item On $[1, 4]$, the maximum value is at the vertex $x=1$: $f(1) = 9$.
\end{itemize}
\[ U_2 = (9 \cdot 3) + (9 \cdot 3) = 54 \]

\subsection*{d. Upper sum with $n=4$ rectangles}
Interval width: $\Delta x = 1.5$. \\
Sub-intervals: $[-2, -0.5], [-0.5, 1], [1, 2.5], [2.5, 4]$.
\begin{itemize}
    \item $[-2, -0.5]$: Max is $f(-0.5) = 6.75$.
    \item $[-0.5, 1]$: Max is $f(1) = 9$.
    \item $[1, 2.5]$: Max is $f(1) = 9$.
    \item $[2.5, 4]$: Max is $f(2.5) = 6.75$.
\end{itemize}
\[ U_4 = 1.5 \times (6.75 + 9 + 9 + 6.75) = 1.5 \times 31.5 = 47.25 \]
